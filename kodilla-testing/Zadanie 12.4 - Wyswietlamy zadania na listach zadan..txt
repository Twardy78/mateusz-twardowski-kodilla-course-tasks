
SELECT IL.NAME, I.SUMMARY
FROM ISSUESLISTS IL
JOIN ISSUES I on IL.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_QTY
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID;

COMMIT;
